The complete file content:

```makefile
#!/usr/bin/make -f
# IgH EtherCAT 1.6 Debian packaging for SeeedStudio Jetson
export DH_VERBOSE = 1

KDIR   := /usr/src/linux-headers-5.15.148-tegra
SRCDIR := ethercat-src
PKGDIR := $(CURDIR)/debian/igh-seeedstudio

%:
	dh $@

# Fetch IgH EtherCAT source from official GitLab (runs before autoreconf/configure)
override_dh_update_autotools_config:
	git clone --depth 1 --branch stable-1.6 \
	    https://gitlab.com/etherlab.org/ethercat.git $(SRCDIR)

# Bootstrap autotools and configure with r8169 + Tegra kernel headers
override_dh_auto_configure:
	cd $(SRCDIR) && ./bootstrap
	cd $(SRCDIR) && ./configure \
	    --prefix=/usr \
	    --sysconfdir=/etc \
	    --with-linux-dir=$(KDIR) \
	    --with-module-dir=extra \
	    --enable-r8169 \
	    --enable-generic \
	    --disable-8139too \
	    --disable-e1000 \
	    --disable-e1000e

# Build userspace + kernel modules, then assert ec_r8169.ko exists
override_dh_auto_build:
	$(MAKE) -C $(SRCDIR) all modules
	@# Fail loudly if r8169 was silently dropped by configure
	test -f $(SRCDIR)/devices/r8169/ec_r8169.ko || \
	    (echo "ERROR: ec_r8169.ko was not built -- check configure output"; exit 1)

# Install userspace tools + kernel modules into package staging
override_dh_auto_install:
	$(MAKE) -C $(SRCDIR) DESTDIR=$(PKGDIR) install
	$(MAKE) -C $(SRCDIR) INSTALL_MOD_PATH=$(PKGDIR) modules_install

# Clean cloned source for rebuild support
override_dh_auto_clean:
	rm -rf $(SRCDIR)

# Skip autoreconf -- we run bootstrap manually in override_dh_auto_configure
override_dh_autoreconf:
	@true

# Exclude .ko files from shared library dependency analysis
override_dh_shlibdeps:
	dh_shlibdeps -X.ko
```

Critical implementation details:
- KDIR must be exactly `/usr/src/linux-headers-5.15.148-tegra` (Tegra kernel headers path)
- SRCDIR is `ethercat-src` (directory name for cloned source)
- PKGDIR uses `$(CURDIR)/debian/igh-seeedstudio` (package staging directory)
- `--prefix=/usr` is a locked decision from STATE.md (ensures ethercatctl reads /etc/ethercat.conf)
- `--with-module-dir=extra` places modules under /lib/modules/5.15.148-tegra/extra/ (matches roadmap success criterion)
- ec_r8169.ko assertion checks `devices/r8169/ec_r8169.ko` (subdirectory layout for kernel 5.15)
- `INSTALL_MOD_PATH` (not DESTDIR) for make modules_install (IgH uses kernel build system convention)
- override_dh_autoreconf with `@true` prevents dh_autoreconf from running in wrong directory
- override_dh_shlibdeps with `-X.ko` prevents dpkg-shlibdeps errors on kernel module files
- File MUST be executable (chmod +x) and use hard tabs for all recipe lines

After writing the file, ensure it is executable: `chmod +x debian/rules`
